fun solve(): Int {
    val (n, m) = readln().split(' ').map(String::toInt)
    val cnt = 32 - (n + m).countLeadingZeroBits()
    var ans = n
    repeat(cnt) {
        if ((n + m) % (1.shl(it)) < 2 * m) {
            ans = ans or 1.shl(it)
        }
    }
    return ans
}

fun solve(): Int {

    var l = n.minus(m).coerceAtLeast(0)
    var r = n + m
    var cnt = 0
    while (l != r) {

        l = l.shr(1)
